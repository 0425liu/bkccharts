     var chartStatic = $("#container").highcharts();

    function addLine(num, title, color) {
        chartStatic.yAxis[0].addPlotLine({
            value: num,
            color: color,
            dashStyle: 'longdashdot',
            width: 2,
            label: {
                text: title + ':' + num,
                align: "right",
                style: {
                    color: color
                },
                x: 0
            },
            zIndex: 20,
        });
    };
});
